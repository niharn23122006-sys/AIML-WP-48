const express = require('express');
const app = express();
const PORT = 3000;

// Home Route
app.get('/', (req, res) => {
    res.send("Welcome to Express Server");
});

// About Route
app.get('/about', (req, res) => {
    res.send("About Page");
});

// 404 Handler
app.use((req, res) => {
    res.status(404).send("404 Page Not Found");
});

app.listen(PORT, () => {
    console.log(`Server running on port ${PORT}`);
});
